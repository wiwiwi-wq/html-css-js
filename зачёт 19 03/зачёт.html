Function Declaration и Function Expression
Практические задания

Задание 1
Объявить функцию через declaration и через expression
// 1. Function Declaration
function greetDeclaration(name) {
    return `Привет, ${name}!`;
}

// 2. Function Expression
const greetExpression = function(name) {
    return `Привет, ${name}!`;
};

// Использование
console.log(greetDeclaration("Анна"));  // Привет, Анна!
console.log(greetExpression("Михаил")); // Привет, Михаил!
Задание 2
Функция-фабрика, возвращающая другую функцию
// Фабрика для создания множителей
function createMultiplier(factor) {
    return function(number) {
        return number * factor;
    };
}

const double = createMultiplier(2);
const triple = createMultiplier(3);

console.log(double(10)); // 20
console.log(triple(10)); // 30

// Фабрика приветствий
function createGreeting(language) {
    return function(name) {
        const greetings = {
            ru: `Здравствуйте, ${name}!`,
            en: `Hello, ${name}!`,
            es: `¡Hola, ${name}!`
        };
        return greetings[language] || `Привет, ${name}!`;
    };
}

const greetRu = createGreeting('ru');
const greetEn = createGreeting('en');

console.log(greetRu('Анна'));  // Здравствуйте, Анна!
console.log(greetEn('John'));  // Hello, John!
